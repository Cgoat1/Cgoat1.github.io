洛谷 P1541 乌龟棋

P1541 乌龟棋

#include<iostream>
using namespace std;
int a[400],b,dp[45][45][45][45];
int main()
{
	int n,m,c[5]={0};
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=1;i<=m;i++)
	{
		cin>>b;
		c[b]++;
	}
	for(int i=0;i<=c[1];i++)
	for(int j=0;j<=c[2];j++)
	for(int k=0;k<=c[3];k++)
	for(int x=0;x<=c[4];x++)
	{
		if(i-1>=0)
			dp[i][j][k][x]=max(dp[i-1][j][k][x],dp[i][j][k][x]);
		if(j-1>=0)
			dp[i][j][k][x]=max(dp[i][j-1][k][x],dp[i][j][k][x]);
		if(k-1>=0)
			dp[i][j][k][x]=max(dp[i][j][k-1][x],dp[i][j][k][x]);
		if(x-1>=0)
			dp[i][j][k][x]=max(dp[i][j][k][x-1],dp[i][j][k][x]);
		dp[i][j][k][x]+=a[i+j*2+k*3+x*4];
	}
	cout<<dp[c[1]][c[2]][c[3]][c[4]]<<endl;
	return 0;
}
