洛谷 P1548棋盘问题

这题由于范围较小(1≤N≤100,1≤M≤100)

可以使用暴力AC

代码如下：

#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j,csquare=0,crectangle=0; //csquare 统计正方形个数  crectangle 统计长方形个数 
	cin>>n>>m; //输入 
	for(i=1;i<=n;i++) //长  可以取任意在n以内的一个数   
	for(j=1;j<=m;j++) //宽  同上
	{
		if(i==j) //如果长等于宽   代表是正方形 
			csquare+=(n-i+1)*(m-j+1); 
		else
			crectangle+=(n-i+1)*(m-j+1); //由于第二遍会重复算长方形，所以不用担心需要乘以二 
		/*
		推导过程：
		总可能性=小长方形的长在大长方形的长中可能性 ×小长方形的宽在大长方形的宽中可能性 
		接下来只用求 长方形的长在大长方形的长中可能性 就可以啦
		我是这样推的： 
			我们先把长方形看成长变成1的小长方形，这个小长方形每次都会往前推一格，直到尽头
			所以，我们只用看它推了多少格就行了
			假设推了n格
			n=大长方形长-小长方形长+1
		然后就没有了...
		*/ 
	} 
	cout<<csquare<<" "<<crectangle<<endl;
 	return 0; //完美结束 q(≧▽≦q) 
}
